## zh-CN

可以通过设置 `treeCheckable` 属性开启复选框勾选。

---

## en-US

The `treeCheckable` property can display checkbox.

---

```vue
<template>
  <div style="marginBottom: 24px;">
    <a-checkbox
      v-model="treeCheckStrictly"
      @change="() => {
        selected = [];
      }"
    >
    treeCheckStrictly
    </a-checkbox>
  </div>
  <a-tree-select
    v-model="selected"
    :allow-search="true"
    :allow-clear="true"
    :tree-checkable="true"
    :tree-check-strictly="treeCheckStrictly"
    :data="treeData"
    placeholder="Please select ..."
    style="width: 300px;"
  ></a-tree-select>
</template>

<script setup>
import { ref } from 'vue';

const selected = ref([]);
const treeCheckStrictly = ref(false);

const treeData = [

    title: 'Trunk 0-0',

    key: '0-0',
    children: [

          {

            key: '0-0-2-1'
          }
        ]
      },
    ],
  },
  {
    title: 'Trunk 0-1',

          {
```
